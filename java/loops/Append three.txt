Problem Statement:- Three digit number n is given. Append to it the digit 3 from the left and from the right.

This problem is from VJudge- Loops- AH

import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int n=sc.nextInt();
		int rev=3;
		while(n>0)
		{
		    int r=n%10;
		    rev=rev*10+r;
		    n=n/10;
		}
		rev=rev*10+3;
		int s=0;
		while(rev>0)
		{
		    int r=rev%10;
		    s=s*10+r;
		    rev=rev/10;
		}
		System.out.println(s);
		
	}
}
