problem statement:-  Find the sum of the first and last digits of a four-digit natural number.

this problem is from VJudge- Loops- AL

import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int n=sc.nextInt();
		int sum=n%10;
		int rev=0;
		while(n>0)
		{
		    int r=n%10;
		    rev=rev*10+r;
		    n=n/10;
		}
		sum=sum+rev%10;
		System.out.println(sum);
	}
}
