import java.util.*;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int N=sc.nextInt();
        int count=0;
        for(int i=1;i<N;i++)
        {
            if((5*i)<=N)
            {
                count++;
            }
            if((3*i)<=N)
            {
                count++;
            }
            if((5*i)>N && (3*i)>N)
            {
                break;
            }
        }
        System.out.println(count);
    }
}
